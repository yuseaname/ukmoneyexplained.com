{{- $content := .Content -}} {{- $faqItems := slice -}} {{- /* Extract FAQ items from h3 headings within the FAQ section */ -}} {{- $inFAQ := false -}} {{- $lines := split $content "\n" -}} {{- $currentQ := "" -}} {{- $currentA := "" -}} {{- range $lines -}} {{- if in . "]+>" "" | replaceRE "^\\s+|\\s+$" "" -}} {{- else if and $inFAQ (ne $currentQ "") -}} {{- $line := . | replaceRE "<[^>]+>" "" | replaceRE "^\\s+|\\s+$" "" -}} {{- if ne $line "" -}} {{- $currentA = printf "%s %s" $currentA $line -}} {{- end -}} {{- end -}} {{- end -}} {{- if and $inFAQ (ne $currentQ "") -}} {{- $faqItems = $faqItems | append (dict "q" $currentQ "a" (trim $currentA "\n ")) -}} {{- end -}} {{- if gt (len $faqItems) 0 -}} {{- end -}}
